13  package de.weltraumschaf.neuron.shell;
14  
15  import de.weltraumschaf.commons.shell.LiteralCommandMap;
16  import de.weltraumschaf.commons.shell.MainCommandType;
17  import de.weltraumschaf.commons.shell.SubCommandType;
18  import java.util.Map;
19  
20  /**
21   * Maps literal strings to keywords.
22   *
23   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
24   */
25  public class NeuronLiteralCommandMap extends LiteralCommandMap {
26  
27      /**
28       * Default constructor.
29       */
30      public NeuronLiteralCommandMap() {
31          super(NeuronSubType.NONE);
32      }
33  
34      @Override
35      protected void initCommandMap(Map<String, MainCommandType> map) {
36          for (final NeuronMainType t : NeuronMainType.values()) {
37              map.put(t.toString(), t);
38          }
39      }
40  
41      @Override
42      protected void initSubCommandMap(Map<String, SubCommandType> map) {
43          for (final NeuronSubType t : NeuronSubType.values()) {
44              if (t.toString().isEmpty()) {
45                  continue; // Ignore to do not recognize empty strings as sub command.
46              }
47              map.put(t.toString(), t);
48          }
49      }
50  
51  }
